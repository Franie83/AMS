System Fingerprint Generator

Run this script on the computer you want to authorize:

# Save as get_fingerprint.py and run
import subprocess
import platform

def get_fingerprint():
    if platform.system() == 'Windows':
        mac = subprocess.check_output('getmac /v /fo list', shell=True).decode()
        for line in mac.split('\n'):
            if 'Physical Address' in line:
                print('MAC:', line.split(':')[1].strip())

        cpu = subprocess.check_output('wmic cpu get processorid', shell=True).decode()
        print('CPU:', cpu.split('\n')[1].strip())

        serial = subprocess.check_output('wmic bios get serialnumber', shell=True).decode()
        print('SERIAL:', serial.split('\n')[1].strip())

if __name__ == '__main__':
    get_fingerprint()
